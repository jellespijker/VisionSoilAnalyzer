47 std::vector<Microscope::Cam_t> Microscope::GetAvailableCams() {
48  const string path_ss = "/sys/class/video4linux";
49  const string path_ss_dev = "/dev/video";
50  std::vector<Cam_t> retVal;
51  struct v4l2_queryctrl queryctrl;
52  struct v4l2_control controlctrl;
53 
54  // Check if there're videodevices installed
55  // Itterate through the cams
56  for (boost::filesystem::directory_iterator itr(path_ss);
57  itr != boost::filesystem::directory_iterator(); ++itr) {
58  string videoln = itr->path().string();
59  videoln.append("/name");
60  if (boost::filesystem::exists(videoln)) {
61  Cam_t currentCam;
62  std::ifstream camName;
63  camName.open(videoln);
64  std::getline(camName, currentCam.Name);
65  camName.close();
66  currentCam.ID =
67  std::atoi(itr->path().string().substr(28, std::string::npos).c_str());
68 
69  // Open Cam
70  currentCam.devString = path_ss_dev + std::to_string(currentCam.ID);
71  if ((currentCam.fd = open(currentCam.devString.c_str(), O_RDWR)) == -1) {
72  throw Exception::MicroscopeException(EXCEPTION_NOCAMS,
73  EXCEPTION_NOCAMS_NR);
74  }
75 
76  // Get controls
77  memset(&queryctrl, 0, sizeof(queryctrl));
78  memset(&controlctrl, 0, sizeof(controlctrl));
79  for (queryctrl.id = V4L2_CID_BASE; queryctrl.id < V4L2_CID_LASTP1;
80  queryctrl.id++) {
81 
82  if (ioctl(currentCam.fd, VIDIOC_QUERYCTRL, &queryctrl) == 0) {
83  if (!(queryctrl.flags & V4L2_CTRL_FLAG_DISABLED)) {
84  Control_t currentControl;
85  currentControl.ID = queryctrl.id;
86  currentControl.name = (char *)queryctrl.name;
87  currentControl.minimum = queryctrl.minimum;
88  currentControl.maximum = queryctrl.maximum;
89  currentControl.default_value = queryctrl.default_value;
90  currentControl.step = queryctrl.step;
91  controlctrl.id = queryctrl.id;
92  if (ioctl(currentCam.fd, VIDIOC_G_CTRL, &controlctrl) == 0) {
93  currentControl.current_value = controlctrl.value;
94  }
95  currentCam.Controls.push_back(currentControl);
96  }
97  } else {
98  if (errno == EINVAL)
99  continue;
100  throw Exception::MicroscopeException(EXCEPTION_QUERY,
101  EXCEPTION_QUERY_NR);
102  }
103  }
104 
105  getResolutions(currentCam, V4L2_PIX_FMT_YUYV);
106  getResolutions(currentCam, V4L2_PIX_FMT_MJPEG);
107  getResolutions(currentCam, V4L2_PIX_FMT_GREY);
108  close(currentCam.fd);
109  retVal.push_back(currentCam);
110  }
111  }
112 
113  for (uint32_t i = 0; i < retVal.size(); i++) {
114  if (retVal[i].Resolutions.size() == 0) {
115  retVal.erase(retVal.begin() + i);
116  i--;
117  }
118  }
119 
120  return retVal;
121 }

123 void Microscope::getResolutions(Cam_t &currentCam, int FormatType) {
124  // Get image formats
125  struct v4l2_format format;
126  memset(&format, 0, sizeof(format));
127 
128  uint32_t width[10] = {640, 800, 1280, 1280, 1920,
129  1600, 2048, 2560, 3840, 3872};
130  uint32_t height[10] = {480, 600, 720, 960, 1080,
131  1200, 1536, 1440, 2160, 2764};
132 
133  uint32_t ResolutionID = 0;
134 
135  for (uint32_t i = 0; i < 10; i++) {
136  format.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
137  format.fmt.pix.pixelformat = FormatType;
138  format.fmt.pix.width = width[i];
139  format.fmt.pix.height = height[i];
140  int ret = ioctl(currentCam.fd, VIDIOC_S_FMT, &format);
141  if (ret != -1 && format.fmt.pix.height == height[i] &&
142  format.fmt.pix.width == width[i]) {
143  Resolution_t res;
144  res.Width = format.fmt.pix.width;
145  res.Height = height[i];
146  res.ID = ResolutionID++;
147  switch (FormatType) {
148  case V4L2_PIX_FMT_YUYV:
149  res.format = PixelFormat::YUYV;
150  break;
151  case V4L2_PIX_FMT_MJPEG:
152  res.format = PixelFormat::MJPG;
153  break;
154  case V4L2_PIX_FMT_GREY:
155  res.format = PixelFormat::GREY;
156  break;
157  default:
158  break;
159  }
160  currentCam.Resolutions.push_back(res);
161  }
162  }
163 }

206  "tisvideobufferfilter", "tisvidebufferfilter");

333 void Microscope::GetHDRFrame(cv::Mat &dst, uint32_t noframes) {
334  // create the brightness steps
335  Control_t *brightness = GetControl("Brightness");
336  Control_t *contrast = GetControl("Contrast");
337 
338  uint32_t brightnessStep =
339  (brightness->maximum - brightness->minimum) / noframes;
340  int8_t currentBrightness = brightness->current_value;
341  int8_t currentContrast = contrast->current_value;
342  contrast->current_value = contrast->maximum;
343 
344  cv::Mat currentImg;
345  // take the shots at different brightness levels
346  for (uint32_t i = 1; i <= noframes; i++) {
347  brightness->current_value = brightness->minimum + (i * brightnessStep);
348  GetFrame(currentImg);
349  HDRframes.push_back(currentImg);
350  }
351 
352  // Set the brightness and back to the previous used level
353  brightness->current_value = currentBrightness;
354  contrast->current_value = currentContrast;
355 
356  // Perform the exposure fusion
357  cv::Mat fusion;
358  cv::Ptr<cv::MergeMertens> merge_mertens = cv::createMergeMertens();
359  merge_mertens->process(HDRframes, fusion);
360  fusion *= 255;
361  fusion.convertTo(dst, CV_8UC1);
362 }
