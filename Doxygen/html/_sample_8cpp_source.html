32 void Sample::PrepImg(SoilSettings *settings) {
33  // setup the settings
34  if (settings == nullptr && Settings == nullptr) {
35  Settings = new SoilSettings;
36  } else {
37  Settings = settings;
38  }
39 
40  // set up the progress signal
41  float currentProg = 0.;
42  prog_sig(currentProg, "Starting segmentation");
43 
44  uint32_t totalsteps = 5;
45  if (Settings->useAdaptiveContrast) {
46  totalsteps++;
47  }
48  if (Settings->useBlur) {
49  totalsteps++;
50  }
51  if (Settings->fillHoles) {
52  totalsteps++;
53  }
54  if (Settings->ignorePartialBorderParticles) {
55  totalsteps++;
56  }
57  if (Settings->morphFilterType != Vision::MorphologicalFilter::NONE) {
58  totalsteps++;
59  }
60  float progstep = 1. / static_cast<float>(totalsteps);
61 
62  if (OriginalImage.empty()) {
63  throw Exception::AnalysisException("No Image found to analyze!", 1);
64  }
65  SHOW_DEBUG_IMG(OriginalImage, uchar, 255, "RGB", false);
66 
67  // Convert the image to an intensity image and an enhanced Intinsity for
68  // better segmentation
69  Vision::Conversion RGBConvertor(OriginalImage);
70  RGBConvertor.Convert(Vision::Conversion::RGB, Vision::Conversion::Intensity);
71  PROG_INCR("Converted to intensity");
72  Intensity = RGBConvertor.ProcessedImg.clone();
73  SHOW_DEBUG_IMG(Intensity, uchar, 255, "Intensity", false);
74 
75  // Enhance the image with an Adaptive contrast stretch and/or followed by a
76  // blur
77  Vision::Enhance IntEnchance(Intensity);
78  if (Settings->useAdaptiveContrast) {
79  IntEnchance.AdaptiveContrastStretch(Settings->adaptContrastKernelSize,
80  Settings->adaptContrastKernelSize);
81  PROG_INCR("Adaptive contrast stretch applied");
82  if (Settings->useBlur) {
83  IntEnchance.Blur(Settings->blurKernelSize, true);
84  PROG_INCR("Blur applied");
85  }
86  } else if (Settings->useBlur) {
87  IntEnchance.Blur(Settings->blurKernelSize, false);
88  PROG_INCR("Blur applied");
89  } else {
90  IntEnchance.ProcessedImg = IntEnchance.OriginalImg;
91  }
92  OptimizedInt =
93  IntEnchance.ProcessedImg(cv::Rect(10, 10, OriginalImage.cols - 20,
94  OriginalImage.rows - 20)).clone();
95  SHOW_DEBUG_IMG(OptimizedInt, uchar, 255, "IntEnchance", false);
96 
97  // Segment the Dark Objects en fill the holes
98  Vision::Segment Segmenter(OptimizedInt);
99  Segmenter.sigma = Settings->sigmaFactor;
100  Segmenter.thresholdOffset = Settings->thresholdOffsetValue;
101  Segmenter.ConvertToBW(Settings->typeOfObjectsSegmented);
102  PROG_INCR("Threshold applied");
103  if (Settings->fillHoles) {
104  Segmenter.FillHoles(true);
105  PROG_INCR("Holes filled");
106  }
107  if (Settings->ignorePartialBorderParticles) {
108  Segmenter.RemoveBorderBlobs(1, true);
109  PROG_INCR("Border Blobs removed");
110  }
111  SHOW_DEBUG_IMG(Segmenter.ProcessedImg, uchar, 255, "Segmenter", true);
112 
113  // Erode the segmented image and sets the BW image use it to create the NO
114  // background RGB
115  Vision::MorphologicalFilter Filter(Segmenter.ProcessedImg);
116  uint kSize = Settings->filterMaskSize;
117  Mat mask = cv::Mat::zeros(kSize, kSize, CV_8UC1);
118  circle(mask, Point(kSize / 2, kSize / 2), (kSize / 2) + 1, 1, -1);
119  switch (Settings->morphFilterType) {
120  case Vision::MorphologicalFilter::CLOSE:
121  Filter.Close(mask);
122  PROG_INCR("Morphological filer - close applied");
123  break;
124  case Vision::MorphologicalFilter::DILATE:
125  Filter.Dilation(mask);
126  PROG_INCR("Morphological filer - dilate applied");
127  break;
128  case Vision::MorphologicalFilter::ERODE:
129  Filter.Erosion(mask);
130  PROG_INCR("Morphological filer - erode applied");
131  break;
132  case Vision::MorphologicalFilter::OPEN:
133  Filter.Open(mask);
134  PROG_INCR("Morphological filer - open applied");
135  break;
136  case Vision::MorphologicalFilter::NONE:
137  Filter.ProcessedImg = Filter.OriginalImg;
138  }
139  BW = Filter.ProcessedImg.clone();
140  SHOW_DEBUG_IMG(BW, uchar, 255,
141  "BW after segmentation, fill holes and erosion", true);
142  RGB = Vision::ImageProcessing::CopyMat<uchar>(
143  OriginalImage(cv::Rect(10, 10, OriginalImage.cols - 20,
144  OriginalImage.rows - 20)).clone(),
145  BW, CV_8UC1);
146  PROG_INCR("RGB masked image generated");
147  SHOW_DEBUG_IMG(RGB, uchar, 255, "RGB no Background", false);
148 
149  // Create the Edge image
150  Vision::Segment Edger(BW);
151  Edger.GetEdgesEroding();
152  Edge = Edger.ProcessedImg;
153  PROG_INCR("Edge filter applied");
154  SHOW_DEBUG_IMG(Edge, uchar, 255, "Edge", true);
155 
156  // Make the CIE La*b* conversion
157  Vision::Conversion RGBnewConvertor(RGB);
158  RGBnewConvertor.Convert(Vision::Conversion::RGB, Vision::Conversion::CIE_lab);
159  LAB = RGBnewConvertor.CopyMat<float>(RGB, BW, CV_32F);
160  PROG_INCR("CIE La*b* conversion calculated");
161  SHOW_DEBUG_IMG(LAB, float, 1.0, "LAB", true);
162 
163  // Create the Redness Index
164  Vision::Conversion LABConvertor(LAB);
165  LABConvertor.Convert(Vision::Conversion::CIE_lab, Vision::Conversion::RI);
166  RI = LABConvertor.ProcessedImg;
167  PROG_INCR("Redness conversion calculated");
168  SHOW_DEBUG_IMG(RI, float, 1.0, "RI", true);
169 
170  imgPrepped = true;
171 }

206 void Sample::SegmentParticles(Vision::Segment::SegmentationType segType) {
207  Vision::Enhance Enhancer(Intensity);
208  Vision::Segment Segmenter;
209  Vision::MorphologicalFilter Eroder;
210 
211  switch (segType) {
212  case Vision::Segment::Normal:
213  // Optimize the intensity image
214  Enhancer.Blur(15);
215  Enhancer.AdaptiveContrastStretch(5, 0.125, true);
216 
217  // Segment the enhance image and remove borderblobs get the edges
218  Segmenter.LoadOriginalImg(Enhancer.ProcessedImg);
219  Segmenter.ConvertToBW(Vision::Segment::Dark);
220  Segmenter.RemoveBorderBlobs(1, true);
221  BW = Segmenter.ProcessedImg.clone();
222  break;
223  case Vision::Segment::LabNeuralNet:
224  throw Exception::AnalysisException("Not yet implemented!", 1);
225  break;
226  case Vision::Segment::GraphMinCut:
227  throw Exception::AnalysisException("Not yet implemented!", 1);
228  break;
229  default:
230  break;
231  }
232 
233  Vision::Segment getEdges(BW);
234  getEdges.GetEdgesEroding();
235  Edge = getEdges.ProcessedImg;
236 
237  // Get the Particlelist
238  Segmenter.GetBlobList(true);
239  Population.resize(Segmenter.BlobList.size());
240  uint32_t i = 0;
241  for_each(Population.begin(), Population.end(), [&](Particle &P) {
242  P.ID = Segmenter.BlobList[i].Label;
243  P.Analysis.Analyzed = false;
244  P.BW = Segmenter.BlobList[i].Img.clone();
245  cv::Rect ROI = Segmenter.BlobList[i].ROI;
246  P.Intensity =
247  Vision::Segment::CopyMat<uchar>(Intensity(ROI).clone(), P.BW, CV_8UC1);
248  P.LAB = Vision::Segment::CopyMat<float>(LAB(ROI).clone(), P.BW, CV_32FC3);
249  P.RGB = Vision::Segment::CopyMat<uchar>(LAB(ROI).clone(), P.BW, CV_8UC3);
250  P.RI = Vision::Segment::CopyMat<float>(LAB(ROI).clone(), P.BW, CV_32FC1);
251  P.Edge = Vision::Segment::CopyMat<uchar>(Edge(ROI).clone(), P.BW, CV_8UC1);
252  i++;
253  });
254 }
