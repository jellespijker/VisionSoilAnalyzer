       C{counter, 3} = bound2im(C{counter,2});
       if (size(C{counter,1}) >= [6 1])

       end

       C{counter, 5} = s.Eccentricity;
       T(counter) = s.Eccentricity;
       counter = counter + 1;
   end
end
